function updateStatus(mode, regime, state, metrics) {

  if (state === "idle") return pick(mode.idle);

  if (state === "firstMotion") return pick(mode.firstMotion);

  if (state === "running") {

    if (metrics.varianceSpike) return mode.reactive.varianceSpike;

    if (metrics.basinMergeEvent) return mode.reactive.basinMerge;

    if (metrics.boundaryFracture) return mode.reactive.boundaryFracture;

    if (metrics.enterCriticality) return mode.reactive.enterCriticality;

    if (metrics.enterChaos) return mode.reactive.enterChaos;

    if (metrics.exitChaos) return mode.reactive.exitChaos;

    if (metrics.approachingStability) return mode.reactive.approachingStability;

    return pick(mode.running);
  }
}
